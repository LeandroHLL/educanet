INSERT INTO matricula (id, cod_curso, cod_modulo, cod_turma, cod_senha)
SELECT ca.id, c.cod_curso, mo.cod_modulo, t.cod_turma, s.cod_senha
FROM cadastro ca
JOIN curso c ON ca.curso = c.cod_curso
JOIN modulo mo ON c.cod_curso = mo.cod_curso
JOIN turma t ON mo.cod_modulo = t.cod_modulo
JOIN senha s ON t.cod_turma = s.cod_turma;

// tabela matricula que ir√° receber os valores de chave estrangeira

CREATE TABLE `matricula` (
  `id` int(11) NOT NULL,
  `cod_curso` int(11) NOT NULL,
  `cod_modulo` int(11) NOT NULL,
  `cod_turma` int(11) NOT NULL,
  `cod_senha` int(11) NOT NULL,
  FOREIGN KEY (`id`) REFERENCES `cadastro` (`id`),
  FOREIGN KEY (`cod_curso`) REFERENCES `curso` (`cod_curso`),
  FOREIGN KEY (`cod_modulo`) REFERENCES `modulo` (`cod_modulo`),
  FOREIGN KEY (`cod_turma`) REFERENCES `turma` (`cod_turma`),
  FOREIGN KEY (`cod_senha`) REFERENCES `senha` (`cod_senha`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_general_ci;
